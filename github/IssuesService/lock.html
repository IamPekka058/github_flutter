Future<void> lock(
  RepositorySlug slug,
  int number, {
  String? lockReason,
}) async {
  String body;
  if (lockReason != null) {
    body = GitHubJson.encode({'lock_reason': lockReason});
  } else {
    body = '{}';
  }
  await github.postJSON(
    '/repos/${slug.fullName}/issues/$number/lock',
    body: body,
    statusCode: 204,
  );
}
