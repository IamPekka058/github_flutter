Stream<Event> listRepositoryIssueEvents(RepositorySlug slug, {int? pages}) {
  return PaginationHelper(github).objects(
    'GET',
    '/repos/${slug.fullName}/issues/events',
    Event.fromJson,
    pages: pages,
  );
